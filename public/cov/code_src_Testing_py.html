30 def computeResult(self, file): 

31 """ 

32 Function to calculate the Error values for a given file and model. 

33 The data is first scaled and then split into Testing sets. 

34 These Testing sets are then predicted by the algorithm and the previsions 

35 are grouped into error windows of two previsions. For each error window an 

36 Error Score is calculated and the window that has the highest error score 

37 is selelted to represent the error score of the whole file. 

38 

39 Args: 

40 file (str): abbreviated filepath 

41 

42 Returns: 

43 numpy Array: Array containing the Error Score of the file 

44 """ 

45 

46 dat_test = self.loader.getDataArray([file]) 

47 dat_test[0] = self.scaler.scale(dat_test[0]) 

48 X, Y = self.algo.prepareData(dat_test, skip=self.config.Prevision_Timesteps) 

49 

50 Y_Pred = self.algo.model.predict(X, verbose=0) 

51 

52 X, Y_Pred = self.algo.pospareData(X, Y_Pred, skip=self.config.Prevision_Timesteps) 

53 X, Y = self.algo.pospareData(X, Y, skip=self.config.Prevision_Timesteps) 

54 # Concatenate prediction data 

55 Y_Pred = self.scaler.invScale(Y_Pred) 

56 Y = self.scaler.invScale(Y) 

57 

58 Y = obj.SimulationFile.createDataFrame(Y, self.config, InputArray=False) 

59 Y = Y.dropna(axis='rows') 

60 Y_Pred = obj.SimulationFile.createDataFrame(Y_Pred, self.config, InputArray=False) 

61 Y_Pred = Y_Pred.dropna(axis='rows') 

62 return Y, Y_Pred
